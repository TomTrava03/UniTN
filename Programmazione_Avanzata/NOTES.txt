Scope Resolution Operator(::) => scope_name::identifier

        std::cout // cout is defined into the namespace std
    Ex. // main.cpp
        #include <iostream>
        using namespace std;

        int x = 3; // GLOBAL VARIABLE
        int main() {
            cout << ::x; // ACCESS GLOBAL VARIABLE EX
